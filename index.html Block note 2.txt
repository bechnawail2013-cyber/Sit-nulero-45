    <script>
        // Variables and state management
        const variables = {
            "nom": "Bechna Wail",
            "age": 22,
            "ville": "Alger",
            "titre": "Mon Projet Algérien",
            "description": "Ceci est un projet créé avec le langage de programmation algérien."
        };
        
        const createdItems = [];
        
        // DOM Elements
        const variablesList = document.getElementById('variablesList');
        const codeInput = document.getElementById('codeInput');
        const previewOutput = document.getElementById('previewOutput');
        const createdItemsGrid = document.getElementById('createdItems');
        const addVariableBtn = document.getElementById('addVariableBtn');
        const variableForm = document.getElementById('variableForm');
        const varNameInput = document.getElementById('varNameInput');
        const varValueInput = document.getElementById('varValueInput');
        const saveVariableBtn = document.getElementById('saveVariableBtn');
        const cancelVariableBtn = document.getElementById('cancelVariableBtn');
        const runCodeBtn = document.getElementById('runCodeBtn');
        const resetCodeBtn = document.getElementById('resetCodeBtn');
        
        // Initialize the variables display
        function updateVariablesDisplay() {
            variablesList.innerHTML = '';
            
            for (const [name, value] of Object.entries(variables)) {
                const variableItem = document.createElement('div');
                variableItem.className = 'variable-item';
                
                const nameSpan = document.createElement('span');
                nameSpan.className = 'variable-name';
                nameSpan.textContent = name;
                
                const valueSpan = document.createElement('span');
                valueSpan.className = 'variable-value';
                
                // Format the value display
                if (typeof value === 'string' && !value.startsWith('"') && !value.startsWith("'")) {
                    valueSpan.textContent = `"${value}"`;
                } else {
                    valueSpan.textContent = value;
                }
                
                // Add delete button
                const deleteBtn = document.createElement('button');
                deleteBtn.innerHTML = '<i class="fas fa-times"></i>';
                deleteBtn.style.background = 'none';
                deleteBtn.style.border = 'none';
                deleteBtn.style.color = 'white';
                deleteBtn.style.cursor = 'pointer';
                deleteBtn.style.marginLeft = '5px';
                
                deleteBtn.addEventListener('click', function(e) {
                    e.stopPropagation();
                    delete variables[name];
                    updateVariablesDisplay();
                });
                
                variableItem.appendChild(nameSpan);
                variableItem.appendChild(valueSpan);
                variableItem.appendChild(deleteBtn);
                variablesList.appendChild(variableItem);
            }
        }
        
        // Initialize the created items display
        function updateCreatedItemsDisplay() {
            // We'll keep the static items for the example
            // In a full implementation, this would update with actual created items
        }
        
        // Show/hide variable form
        addVariableBtn.addEventListener('click', function() {
            variableForm.style.display = 'block';
            addVariableBtn.style.display = 'none';
            varNameInput.focus();
        });
        
        cancelVariableBtn.addEventListener('click', function() {
            variableForm.style.display = 'none';
            addVariableBtn.style.display = 'inline-flex';
            varNameInput.value = '';
            varValueInput.value = '';
        });
        
        // Save new variable
        saveVariableBtn.addEventListener('click', function() {
            const name = varNameInput.value.trim();
            let value = varValueInput.value.trim();
            
            if (!name) {
                alert('Veuillez entrer un nom de variable');
                return;
            }
            
            // Try to parse numbers
            if (!isNaN(value) && value !== '') {
                value = Number(value);
            }
            // Check if it's a string (in quotes)
            else if ((value.startsWith('"') && value.endsWith('"')) || 
                    (value.startsWith("'") && value.endsWith("'"))) {
                value = value.slice(1, -1);
            }
            
            variables[name] = value;
            updateVariablesDisplay();
            
            variableForm.style.display = 'none';
            addVariableBtn.style.display = 'inline-flex';
            varNameInput.value = '';
            varValueInput.value = '';
        });
        
        // Run code execution
        runCodeBtn.addEventListener('click', executeCode);
        
        // Reset code
        resetCodeBtn.addEventListener('click', function() {
            codeInput.value = '';
            previewOutput.innerHTML = `
                <div style="text-align: center; padding: 40px 20px; color: var(--gray);">
                    <i class="fas fa-code" style="font-size: 3rem; margin-bottom: 20px; opacity: 0.5;"></i>
                    <p>Exécutez votre code pour voir la prévisualisation ici</p>
                </div>
            `;
        });
        
        // Example cards click handlers
        document.querySelectorAll('.example-card').forEach(card => {
            card.addEventListener('click', function() {
                const exampleType = this.getAttribute('data-example');
                loadExample(exampleType);
            });
        });
        
        // Load example code
        function loadExample(exampleType) {
            let exampleCode = '';
            
            switch(exampleType) {
                case 'flag':
                    exampleCode = `créer drapeau('Algérie', 'vert', 'blanc', 'rouge')
créer drapeau('Maroc', 'rouge', 'vert')
créer drapeau('Tunisie', 'rouge', 'blanc')`;
                    break;
                    
                case 'card':
                    exampleCode = `créer carte('Alger', 'Capitale de l\\'Algérie, située sur la côte méditerranéenne.')
créer carte('Oran', 'Deuxième ville d\\'Algérie, importante métropole économique.')
créer carte('Constantine', 'Ville historique connue pour ses ponts suspendus.')`;
                    break;
                    
                case 'button':
                    exampleCode = `créer bouton('Cliquez ici', 'primary')
créer bouton('Sauvegarder', 'success')
créer bouton('Annuler', 'warning')
créer bouton('Supprimer', 'error')`;
                    break;
                    
                case 'variables':
                    exampleCode = `définir pays = 'Algérie'
définir population = 44
définir capitale = 'Alger'
définir langues = ['Arabe', 'Français', 'Berbère']

afficher 'Pays: ' + pays
afficher 'Population: ' + population + ' millions'
afficher 'Capitale: ' + capitale
afficher 'Langues: ' + langues`;
                    break;
            }
            
            codeInput.value = exampleCode;
        }
        
        // Main code execution function
        function executeCode() {
            const code = codeInput.value;
            const lines = code.split('\n');
            
            // Clear previous output
            previewOutput.innerHTML = '';
            
            // Process each line
            for (let line of lines) {
                line = line.trim();
                if (!line) continue;
                
                // Handle variable definition: définir nom = valeur
                if (line.startsWith('définir ') || line.startsWith('definir ')) {
                    const varMatch = line.match(/d[eé]finir\s+(\w+)\s*=\s*(.+)/);
                    if (varMatch) {
                        const varName = varMatch[1];
                        let varValue = varMatch[2].trim();
                        
                        // Remove quotes if present
                        if ((varValue.startsWith('"') && varValue.endsWith('"')) || 
                            (varValue.startsWith("'") && varValue.endsWith("'"))) {
                            varValue = varValue.slice(1, -1);
                        }
                        // Try to parse as number
                        else if (!isNaN(varValue)) {
                            varValue = Number(varValue);
                        }
                        
                        variables[varName] = varValue;
                    }
                }
                // Handle display command: afficher valeur
                else if (line.startsWith('afficher ')) {
                    const displayMatch = line.match(/afficher\s+(.+)/);
                    if (displayMatch) {
                        let displayValue = displayMatch[1].trim();
                        
                        // Check if it's a variable
                        if (variables[displayValue] !== undefined) {
                            displayValue = variables[displayValue];
                        }

        // Corriger automatiquement le code
        function autoCorrectCode(code) {
            let corrections = [];
            let correctedCode = code;
            
            // Détecter et corriger les erreurs courantes
            if (code.includes('prit(') || code.includes('pritn(')) {
                correctedCode = correctedCode.replace(/prit(?:n)?\(/g, 'print(');
                corrections.push('"prit" ou "pritn" corrigé en "print"');
            }
            
            if (code.includes('repeet') || code.includes('repete')) {
                correctedCode = correctedCode.replace(/repeet|repete/g, 'repeat');
                corrections.push('"repeet" ou "repete" corrigé en "repeat"');
            }
            
            if (code.includes('repet')) {
                correctedCode = correctedCode.replace(/repet/g, 'repeat');
                corrections.push('"repet" corrigé en "repeat"');
            }
            
            // Vérifier les guillemets non fermés
            const quoteCount = (code.match(/"/g) || []).length;
            if (quoteCount % 2 !== 0) {
                correctedCode += '"';
                corrections.push('Guillemet fermant ajouté');
            }
            
            return {
                corrected: correctedCode,
                corrections: corrections
            };
        }

                        repeatCount = parseInt(repeatMatch[1]);
                    }

                        output += text + '\n';
